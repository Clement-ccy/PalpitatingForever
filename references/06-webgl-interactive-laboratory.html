Chaos Attractor

A particle simulation exploring strange attractors and non-linear dynamics. Interact with the field using your cursor to disrupt the flow.

THREE.JS WEBGL GLSL
fragment.glsl
void main() {
vec2 uv = gl_FragCoord.xy / u_resolution;
float t = u_time * 0.5;
// Create noise field
for(float i = 1.0; i < 4.0; i++) {
uv.x += 0.6 / i * cos(i * 2.5 * uv.y + t);
uv.y += 0.6 / i * cos(i * 1.5 * uv.x + t);
}
vec3 col = vec3(0.5 * sin(t) + 0.5, uv.x, uv.y);
gl_FragColor = vec4(col, 1.0);
}
